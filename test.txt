package net.ogalab.akkacookbook.ch01.sec003;

import akka.actor.ActorRef;
import akka.actor.ActorSystem;

/** Defining the actor's behavior and state.
 *
 * @author oogasawa
 */
public class BehaviorAndStateApp {
    
    public static void main(String[] args) throws InterruptedException {
        final ActorSystem system = ActorSystem.create("BehaviorAndStateApp");
        
        final ActorRef actor = system.actorOf(SummingActor.props(), "summingactor");

        System.out.println(actor.path());
        

        final ActorRef actor2 = system.actorOf(SummingActorWithConstructor.props(10), "summingactor2");
        actor2.tell(Integer.valueOf(3), ActorRef.noSender());
        actor2.tell(Integer.valueOf(5), ActorRef.noSender());
        actor2.tell("What's up?", ActorRef.noSender());

        System.out.println(actor2.path());
        
        Thread.sleep(5000);
        system.terminate();
    }
    
}
